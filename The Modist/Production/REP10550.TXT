                      ERROR(Text1040018);
                    IF ClientFile = '' THEN
                      ClientFile := FileMgt.SaveFileDialog(Text1040026,ExtensionTxt,'');

                    IF ISCLEAR(FSAuto) THEN
                      CREATE(FSAuto,TRUE,TRUE);
                    BACSFile.TEXTMODE(TRUE);
                    BACSFile.WRITEMODE(TRUE);
                    IF FILE.EXISTS(ClientFile) THEN BEGIN
                      IF ExpDialog.RUNMODAL IN [ACTION::OK,ACTION::Yes] THEN BEGIN
                        ExpDialog.GetSelectedOption(Selection);
                        CLEAR(ExpDialog);
                        CASE Selection OF
                          Selection::Overwrite:
                            Mode := Text1040019;
                          Selection::Append:
                            Mode := Text1040020;
                        END;
                        BACSFile.CREATE(ServerFilename);
                      END ELSE
                        ERROR('');
                    END ELSE
                      BACSFile.CREATE(ServerFilename);
                  END;

                  IF  ExportBACS THEN BEGIN
                    Settings := STRSUBSTNO(Text1040021,ClientFile,FORMAT(TODAY),FORMAT(TIME),Mode)
                  END ELSE
                    Settings := Text1040022;
                END;

    OnPostReport=VAR
                   FileMgt@1040000 : Codeunit 419;
                 BEGIN
                   IF ExportBACS THEN BEGIN
                     BACSFile.CLOSE;
                     IF PurchSetup."Import BACS Files" THEN BEGIN
                       BACSRegister.GET(BACSManagement.GetRegisterNo);
                       BACSRegister.ImportBACSFile(ServerFilename,FALSE);
                       BACSRegister.MODIFY;
                     END;
                     PurchSetup.GET;
                     CASE Selection OF
                       Selection::Overwrite:
                         FileMgt.DownloadToFile(ServerFilename,ClientFile);
                       Selection::Append:
                         FileMgt.AppendAllTextToClientFile(ServerFilename,ClientFile);
                     END;
                     MESSAGE(Text1040023);
                     FILE.ERASE(ServerFilename);
                     UpdateDefaultBACSFileName(ClientFile);
                   END;
                 END;
