    LOCAL PROCEDURE SetTransportFilter@9(TransportNo@1000 : Code[20]);

    LOCAL PROCEDURE SetOutdatedFilter@10();
    VAR
      WhereUsedStatus@1000 : Record 11102059;
      Object@1001 : Record 2000000001;
      SourceControl@1002 : Codeunit 11102050;
      UseWhereUsedStatus@1003 : Boolean;
    BEGIN
      CLEARMARKS;
      MARKEDONLY(FALSE);

      CASE SourceControl.SqlsAnswer OF
        '0' : UseWhereUsedStatus := TRUE;
        '1' :
          BEGIN
            MARKEDONLY(TRUE);
            EXIT;
          END;
      END;

      IF UseWhereUsedStatus THEN BEGIN
        IF WhereUsedStatus.FINDSET THEN
          REPEAT
            IF Object.GET(WhereUsedStatus."Object Type", '', WhereUsedStatus."Object No.") THEN
              IF SourceControl.ObjectHeaderIsOutdated(Object, WhereUsedOutdated) THEN
                MarkRecord(Object.Type, Object.ID);
          UNTIL WhereUsedStatus.NEXT = 0;
      END ELSE
        IF Object.FINDSET THEN
          REPEAT
            IF SourceControl.ObjectHeaderIsOutdated(Object, WhereUsedOutdated) THEN
              MarkRecord(Object.Type, Object.ID);
          UNTIL Object.NEXT = 0;

      SetMarkedOnly;
    END;
