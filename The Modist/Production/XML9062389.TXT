    LOCAL PROCEDURE AddCriteria@1000000006(CriteriaId@1000000000 : Integer);
    VAR
      SrcCriteria@1000000001 : Record 9062626;
      SrcCriteriaField@1000000002 : Record 9062627;
    BEGIN
      IF (CriteriaId = 0) THEN
        EXIT;
      IF (dCriteria.GET(CriteriaId)) THEN
        EXIT;

      SrcCriteria.GET(CriteriaId);
      dCriteria.TRANSFERFIELDS(SrcCriteria);
      dCriteria.INSERT();

      SrcCriteriaField.RESET();
      SrcCriteriaField.SETRANGE("Criteria ID", SrcCriteria.ID);
      IF (SrcCriteriaField.FINDSET()) THEN BEGIN
        REPEAT
          dCriteriaField.TRANSFERFIELDS(SrcCriteriaField);
          dCriteriaField.INSERT();
        UNTIL (SrcCriteriaField.NEXT() = 0);
      END;
    END;

    LOCAL PROCEDURE ProcessImportedCriterias@1000000012();
    VAR
      DestCriteria@1000000000 : Record 9062626;
      DestCriteriaField@1000000001 : Record 9062627;
    BEGIN
      dCriteria.RESET();
      IF (dCriteria.FINDSET()) THEN BEGIN
        REPEAT
          DestCriteria.TRANSFERFIELDS(dCriteria);
          DestCriteria.ID := 0;
          DestCriteria.INSERT();

          dCriteriaField.RESET();
          dCriteriaField.SETRANGE("Criteria ID", dCriteria.ID);
          IF (dCriteriaField.FINDSET()) THEN BEGIN
            DestCriteriaField.TRANSFERFIELDS(dCriteriaField);
            DestCriteriaField.ID := 0;
            DestCriteriaField."Criteria ID" := DestCriteria.ID;
            DestCriteriaField.INSERT();
          END;

          ImportedLine.RESET();
          ImportedLine.SETRANGE("Filter Criteria ID", dCriteria.ID);
          IF (ImportedLine.FINDSET()) THEN BEGIN
            REPEAT
              dRecordBrowserLine.GET(ImportedLine."Record Browser Code", ImportedLine."Line No.");
              dRecordBrowserLine."Filter Criteria ID" := DestCriteria.ID;
              dRecordBrowserLine.MODIFY();
            UNTIL (ImportedLine.NEXT() = 0);
          END;

          ImportedLine.RESET();
          ImportedLine.SETRANGE("Link Criteria ID", dCriteria.ID);
          IF (ImportedLine.FINDSET()) THEN BEGIN
            REPEAT
              dRecordBrowserLine.GET(ImportedLine."Record Browser Code", ImportedLine."Line No.");
              dRecordBrowserLine."Link Criteria ID" := DestCriteria.ID;
              dRecordBrowserLine.MODIFY();
            UNTIL (ImportedLine.NEXT() = 0);
          END;

          ImportedLineField.RESET();
          ImportedLineField.SETRANGE("Visibility Criteria ID", dCriteria.ID);
          IF (ImportedLineField.FINDSET()) THEN BEGIN
            REPEAT
              dRecordBrowserLineField.GET(ImportedLineField."Record Browser Code",
                ImportedLineField."Line Type",
                ImportedLineField."Record Browser Line No.",
                ImportedLineField."Line No.");
              dRecordBrowserLineField."Visibility Criteria ID" := DestCriteria.ID;
              dRecordBrowserLineField.MODIFY();
            UNTIL (ImportedLineField.NEXT() = 0);
          END;

        UNTIL (dCriteria.NEXT() = 0);
      END;
    END;

    BEGIN

    }
    END.
