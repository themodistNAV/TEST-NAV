    PROCEDURE SetEndingDate@1102();
    BEGIN
      AccountingPeriod.RESET;
      AccountingPeriod.SETFILTER("Period Start",'>%1',PeriodStartingDate);
      AccountingPeriod.FIND('-');
      PeriodEndingDate := AccountingPeriod."Period Start" - 1;
    END;

    PROCEDURE SetPreviousDates@1103();
    BEGIN
      PreviousStartingDate := CALCDATE(Text006,PeriodStartingDate);
      PreviousEndingDate := CALCDATE(Text006,PeriodEndingDate);
      CheckDates;
    END;

    LOCAL PROCEDURE CheckDates@1104();
    VAR
      AccountingPeriod2@1105 : Record 10560;
      AccountingPeriod3@1106 : Record 10560;
    BEGIN
      AccountingPeriod2.SETRANGE("Period Type",AccountingPeriod2."Period Type"::Year);
      AccountingPeriod2.SETFILTER("Period Start",'<=%1',PeriodStartingDate);
      AccountingPeriod2.FIND('+');

      AccountingPeriod3.SETRANGE("Period Type",AccountingPeriod3."Period Type"::Year);
      AccountingPeriod3.SETFILTER("Period Start",'<=%1',PeriodEndingDate);
      AccountingPeriod3.FIND('+');
      IF AccountingPeriod2."Period Start" <> AccountingPeriod3."Period Start" THEN
        ERROR(
          STRSUBSTNO(Text014),
            PeriodStartingDate,PeriodEndingDate);

      AccountingPeriod2.RESET;
      AccountingPeriod3.RESET;

      AccountingPeriod2.SETRANGE("Period Type",AccountingPeriod2."Period Type"::Year);
      AccountingPeriod2.SETFILTER("Period Start",'<=%1',PreviousStartingDate);
      AccountingPeriod2.FIND('+');

      AccountingPeriod3.SETRANGE("Period Type",AccountingPeriod2."Period Type"::Year);
      AccountingPeriod3.SETFILTER("Period Start",'<=%1',PreviousEndingDate);
      AccountingPeriod3.FIND('+');
      IF AccountingPeriod2."Period Start" <> AccountingPeriod3."Period Start" THEN
        ERROR(
          STRSUBSTNO(Text014),
            PreviousStartingDate,PreviousEndingDate);
    END;

    PROCEDURE RoundAmount@1107(Value@1107 : Decimal) : Text[30];

    END;

    PROCEDURE InitializeRequest@2(NewPeriodStartingDate@1000 : Date;NewPeriodEndingDate@1001 : Date;NewPreviousStartingDate@1002 : Date;NewPreviousEndingDate@1003 : Date;NewRoundingFactor@1004 : Option;NewIndent@1005 : Option);

      RoundingFactor := NewRoundingFactor;
      IF NewIndent <> Indent::None THEN BEGIN
        Indent := NewIndent;
        CheckIndent;
      END;
    END;

    LOCAL PROCEDURE CheckIndent@4();
    BEGIN
      GLIndent.RESET;
      MaxIndent := '';
      IF GLIndent.FINDSET THEN
        REPEAT
          IF FORMAT(GLIndent.Indentation) > MaxIndent THEN
            MaxIndent := FORMAT(GLIndent.Indentation);
        UNTIL GLIndent.NEXT = 0;

      IF FORMAT(Indent) > MaxIndent THEN BEGIN
        IF Indent <> Indent::None THEN
          ERROR(STRSUBSTNO(Text015,SELECTSTR(Indent + 1,IndentTxt),MaxIndent));
      END;
    END;

    BEGIN
    END.
