    PROCEDURE CustomGet@1();
    VAR
      Object@1000 : Record 2000000001;
      Initialize@1001 : Codeunit 11102041;
    BEGIN
      IF NOT GET(PersonalMachineCode) THEN
        IF NOT GET(PersonalCode) THEN
          IF NOT GET THEN BEGIN
            INSERT;
            Initialize.Initialize(0);
            GET;
          END;
      IF "Data Convert to Version No." < OMAVersionNo THEN
        IF Object.GET(Object.Type::Codeunit, '', CODEUNIT::"OM - Initialize") THEN
          IF Object."Version List" = OMAVersionList THEN
            Initialize.DataConversion;
    END;

    PROCEDURE PersonalCode@18() : Text;
    VAR
      Mgt@1000 : Codeunit 11102035;
    BEGIN
      EXIT(Mgt.ShortUserId);
    END;

    PROCEDURE PersonalMachineCode@19() ExitValue : Text;
    VAR
      Mgt@1000 : Codeunit 11102035;
      ApplicationSettings@1001 : Codeunit 11102101;
    BEGIN
      ExitValue := ApplicationSettings.ComputerName + '\' + Mgt.ShortUserId;
      IF STRLEN(ExitValue) > MAXSTRLEN("Primary Key") THEN
        ExitValue := COPYSTR(ApplicationSettings.ComputerName, 1, MAXSTRLEN("Primary Key") - STRLEN(Mgt.ShortUserId) - 1) + '\' + Mgt.ShortUserId;
    END;

    LOCAL PROCEDURE SetGlobalSetting@2(ChangeFieldNo@1000 : Integer;DoMessage@1001 : Boolean);
    VAR
      Setup@1002 : Record 11102035;
      ctSettingCopied@1003 : TextConst 'DEU=This setting applies to all users.;ENU=This setting applies to all users.;ESP=This setting applies to all users.;FRA=This setting applies to all users.;NLD=Deze instelling geldt voor alle gebruikers.;DEA=This setting applies to all users.';
    BEGIN
      Setup.SETFILTER("Primary Key", '<>%1', "Primary Key");
      IF Setup.ISEMPTY THEN
        EXIT;
      CASE ChangeFieldNo OF
        FIELDNO("Role Shortcut 1") :
          Setup.MODIFYALL("Role Shortcut 1", "Role Shortcut 1");
        FIELDNO("Role Shortcut 2") :
          Setup.MODIFYALL("Role Shortcut 2", "Role Shortcut 2");
        FIELDNO("Role Shortcut 3") :
          Setup.MODIFYALL("Role Shortcut 3", "Role Shortcut 3");
        FIELDNO("Role Shortcut 4") :
          Setup.MODIFYALL("Role Shortcut 4", "Role Shortcut 4");
        FIELDNO("Role Shortcut 5") :
          Setup.MODIFYALL("Role Shortcut 5", "Role Shortcut 5");
      END;
      IF DoMessage THEN
        MESSAGE(ctSettingCopied);
    END;

    LOCAL PROCEDURE CheckTraceTrigger@8();
    VAR
      SQLMgt@1000 : Codeunit 11102075;
      ctInstallTrigger@1001 : TextConst 'DEU=The Trace Trigger is needed to use this option.\Do you want to install it now?;ENU=The Trace Trigger is needed to use this option.\Do you want to install it now?;ESP=The Trace Trigger is needed to use this option.\Do you want to install it now?;FRA=The Trace Trigger is needed to use this option.\Do you want to install it now?;NLD=De traceer-trigger is nodig voor deze functie.\Wilt u deze nu installeren?;DEA=The Trace Trigger is needed to use this option.\Do you want to install it now?';
    BEGIN
      SQLMgt.RUN;
      IF SQLMgt.GetTraceTriggerStatus <> 3 THEN
        IF CONFIRM(ctInstallTrigger, TRUE) THEN
          SQLMgt.AddTraceTrigger(TRUE);
      IF SQLMgt.GetTraceTriggerStatus <> 3 THEN
        ERROR('');
    END;
