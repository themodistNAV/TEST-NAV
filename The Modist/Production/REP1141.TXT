OBJECT Report 1141 Delete Old Cost Entries
{
  OBJECT-PROPERTIES
  {
    Date=07/09/12;
    Time=12:00:00 PM;
    Version List=NAVW17.00;
  }
  PROPERTIES
  {
    CaptionML=[ENU=Delete Old Cost Entries;
               ENG=Delete Old Cost Entries];
    ProcessingOnly=Yes;
  }
  DATASET
  {
  }
  REQUESTPAGE
  {
    PROPERTIES
    {
    }
    CONTROLS
    {
      { 1   ;    ;Container ;
                  ContainerType=ContentArea }

      { 2   ;1   ;Field     ;
                  Name=YearEndingDate;
                  CaptionML=[ENU=Year Ending Date;
                             ENG=Year Ending Date];
                  SourceExpr=YearEndDate;
                  OnValidate=BEGIN
                               IF YearEndDate <> CALCDATE('<CY>',YearEndDate) THEN
                                 ERROR(Text001,YearEndDate);

                               IF WORKDATE - YearEndDate < 365 THEN
                                 ERROR(Text002,YearEndDate);

                               IF NOT CONFIRM(Text003,FALSE,YearEndDate) THEN
                                 EXIT;

                               CostEntry.SETCURRENTKEY("Cost Type No.","Posting Date");
                               CostEntry.SETRANGE("Posting Date",0D,YearEndDate);
                               IF NOT CostEntry.ISEMPTY THEN BEGIN
                                 CostEntry.DELETEALL;
                                 MESSAGE(Text004,YearEndDate);
                               END ELSE
                                 ERROR(Text005,YearEndDate);
                             END;
                              }

    }
  }
  LABELS
  {
  }
  CODE
  {
    VAR
      CostEntry@1006 : Record 1104;
      YearEndDate@1000 : Date;
      Text001@1005 : TextConst 'ENU=%1 is not at year''s end.;ENG=%1 is not at year''s end.';
      Text002@1004 : TextConst 'ENU=The selected year ending date %1 must be older than last year.;ENG=The selected year ending date %1 must be older than last year.';
      Text003@1003 : TextConst 'ENU=Are you sure you want to delete all cost entries up to and including %1?;ENG=Are you sure you want to delete all cost entries up to and including %1?';
      Text004@1002 : TextConst 'ENU=All cost entries up to and including %1 deleted.;ENG=All cost entries up to and including %1 deleted.';
      Text005@1001 : TextConst 'ENU=No cost entries were found before %1.;ENG=No cost entries were found before %1.';

    BEGIN
    END.
  }
