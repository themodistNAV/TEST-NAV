    LOCAL PROCEDURE PerformColumnsAutoSizing@1000000009();
    VAR
      lFieldIndex@1000000000 : Integer;
      lUsedPaperWidth@1000000001 : Decimal;
    BEGIN
      lUsedPaperWidth := 0;
      FOR lFieldIndex := 1 TO Fields DO

      IF (lUsedPaperWidth > PaperWidth) THEN

    END;

    LOCAL PROCEDURE PerformColumnsAutoShrinking@1000000003(pUsedPaperWidth@1000000004 : Decimal);
    VAR
      lShrinking@1000000000 : Boolean;
      lShrinkingMade@1000000001 : Boolean;
      lFieldIndex@1000000007 : Integer;
      lCurrentColumnSize@1000000005 : Integer;
      lPossibleColumnSize@1000000006 : Integer;
      lWidthDelta@1000000008 : Decimal;
    BEGIN
      lShrinking := TRUE;
      WHILE (lShrinking) DO
      BEGIN
        lShrinkingMade := FALSE;
        FOR lFieldIndex := 1 TO Fields DO
        BEGIN
          IF (pUsedPaperWidth > PaperWidth) THEN
          BEGIN
            lCurrentColumnSize := FieldColumnSizes[lFieldIndex];
            lPossibleColumnSize := ColumnSizes[FieldDataTypes[lFieldIndex]][ColumnSizeIndex::Min];
            IF (lCurrentColumnSize > lPossibleColumnSize) THEN
            BEGIN
              lWidthDelta := ColumnWidths[lCurrentColumnSize] - ColumnWidths[lPossibleColumnSize];
              IF (pUsedPaperWidth - lWidthDelta >= PaperWidth) THEN
              BEGIN
                pUsedPaperWidth -= lWidthDelta;
                FieldColumnSizes[lFieldIndex] -= 1;
                lShrinkingMade := TRUE;
              END;
            END;
          END
          ELSE
          BEGIN
            lFieldIndex := Fields;
            lShrinking := FALSE;
          END;
        END;
        IF ( NOT (lShrinkingMade)) THEN
          lShrinking := FALSE;
      END;
    END;
