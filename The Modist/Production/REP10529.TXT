    LOCAL PROCEDURE FormatAmt@15(AmountToPrint@1040000 : Decimal) : Text;

      TextAmt@1000 : Text;
    BEGIN
      TextAmt := FORMAT(ROUND(-AmountToPrint,1),0,'<Integer Thousand><Decimals>');

        TextAmt := '(' + TextAmt + ')';
      EXIT(TextAmt);
    END;

    PROCEDURE IncrLineCount@1();

    END;
