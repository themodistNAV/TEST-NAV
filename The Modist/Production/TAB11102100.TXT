    PROCEDURE SetMyObjectsFilter@14();
    VAR
      User@1000 : Record 11102056;
      Object@1001 : Record 2000000001;
      TableFilterMgt@1002 : Codeunit 11102100;
      RecRef@1003 : RecordRef;
    BEGIN
      SETFILTER("My Type Filter", '<>0');

      User.GetActiveUser(FALSE);
      IF FORMAT(User."My Objects Filter") IN ['', TableFilterMgt.EmptyTableFilter(DATABASE::Object)] THEN
        EXIT;

      Object.SETFILTER(Type, '<>%1', Object.Type::TableData);

      RecRef.GETTABLE(Object);
      TableFilterMgt.FilterRecRef(RecRef, FORMAT(User."My Objects Filter"), 0);
      RecRef.SETTABLE(Object);

      Object.COPYFILTER(Type, "My Type Filter");
      Object.COPYFILTER(ID, "My No. Filter");

      IF Object.GETFILTER(Name) = '' THEN
        SETRANGE("My Name Filter")
      ELSE
        SETFILTER("My Name Filter", Object.GETFILTER(Name));

      Object.COPYFILTER(Modified, "My Modified Filter");
      Object.COPYFILTER(Date, "My Date Filter");
      Object.COPYFILTER(Time, "My Time Filter");

      IF Object.GETFILTER("Version List") = '' THEN
        SETRANGE("My Version List Filter")
      ELSE
        SETFILTER("My Version List Filter", Object.GETFILTER("Version List"));

      IF Object.GETFILTER(Caption) = '' THEN
        SETRANGE("My Caption Filter")
      ELSE
        SETFILTER("My Caption Filter", Object.GETFILTER(Caption));
    END;
