    LOCAL PROCEDURE FindPurchaseDocument@1170000000(DocumentTypeP@1170000000 : 'Quote,Order,Invoice,Credit Memo,Blanket Order,Return Order');
    BEGIN
      PurchaseHeaderArchiveG.SETRANGE("Document Type", DocumentTypeP);
      PurchaseHeaderArchiveG.SETRANGE("No.", DocumentCostCenter."Document No.");
      IF PurchaseHeaderArchiveG.FINDLAST THEN BEGIN
        SeasonCodeG := PurchaseHeaderArchiveG."Season Code";
        SupplierG := PurchaseHeaderArchiveG."Buy-from Vendor Name";
        CurrencyCodeG := PurchaseHeaderArchiveG."Currency Code";
        DescriptionG := PurchaseHeaderArchiveG."Posting Description";
        PostingDateG := PurchaseHeaderArchiveG."Posting Date";
        TotalCurrencyLCYG := CurrencyExchangeRateG.ExchangeAmtFCYToLCY(PurchaseHeaderArchiveG."Posting Date", PurchaseHeaderArchiveG."Currency Code",
                                                                        DocumentCostCenter.Amount, PurchaseHeaderArchiveG."Currency Factor");
        PurchaseLineArchiveG.SETRANGE("Document Type", PurchaseHeaderArchiveG."Document Type");
        PurchaseLineArchiveG.SETRANGE("Document No.", PurchaseHeaderArchiveG."No.");
        PurchaseLineArchiveG.SETRANGE(Type, PurchaseLineArchiveG.Type::"G/L Account");
        PurchaseLineArchiveG.SETRANGE("Line No.", DocumentCostCenter."Document Line No.");
        IF PurchaseLineArchiveG.FINDFIRST THEN BEGIN
          GLAccountNoG := PurchaseLineArchiveG."No.";
          GLAccountNameG := PurchaseLineArchiveG.Description;
        END;
      END ELSE BEGIN
        PurchInvoiceHdrG.SETCURRENTKEY("Order No.");
        PurchInvoiceHdrG.SETRANGE("Order No.", DocumentCostCenter."Document No.");
        IF NOT PurchInvoiceHdrG.FINDFIRST THEN BEGIN
          PurchInvoiceHdrG.SETRANGE("Order No.");
          PurchInvoiceHdrG.SETRANGE("Pre-Assigned No.", DocumentCostCenter."Document No.");
          PurchInvoiceHdrG.FINDFIRST;
        END;
        SeasonCodeG := PurchInvoiceHdrG."Season Code";
        SupplierG := PurchInvoiceHdrG."Buy-from Vendor Name";
        CurrencyCodeG := PurchInvoiceHdrG."Currency Code";
        DescriptionG := PurchInvoiceHdrG."Posting Description";
        PostingDateG := PurchInvoiceHdrG."Posting Date";
        TotalCurrencyLCYG := CurrencyExchangeRateG.ExchangeAmtFCYToLCY(PurchInvoiceHdrG."Posting Date", PurchInvoiceHdrG."Currency Code",
                                                                         DocumentCostCenter.Amount,PurchInvoiceHdrG."Currency Factor");
        PurchInvoiceLineG.SETRANGE("Document No.", PurchInvoiceHdrG."No.");
        PurchInvoiceLineG.SETRANGE(Type, PurchInvoiceLineG.Type::"G/L Account");
        PurchInvoiceLineG.SETRANGE("Line No.", DocumentCostCenter."Document Line No.");
        IF PurchInvoiceLineG.FINDFIRST THEN BEGIN
          GLAccountNoG := PurchInvoiceLineG."No.";
          GLAccountNameG := PurchInvoiceLineG.Description;
        END;
      END;
    END;
