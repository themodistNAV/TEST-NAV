OBJECT Page 9901 Export Data
{
  OBJECT-PROPERTIES
  {
    Date=15/09/15;
    Time=12:00:00 PM;
    Version List=NAVW19.00;
  }
  PROPERTIES
  {
    CaptionML=[ENU=Export Data;
               ENG=Export Data];
    InsertAllowed=No;
    DeleteAllowed=No;
    ModifyAllowed=No;
    SourceTable=Table2000000006;
    PageType=StandardDialog;
    SourceTableTemporary=Yes;
    OnOpenPage=VAR
                 Company@1000 : Record 2000000006;
               BEGIN
                 IncludeApplication := FALSE;
                 IncludeApplicationData := FALSE;
                 IncludeGlobalData := TRUE;
                 IncludeAllCompanies := TRUE;

                 IF Company.FINDSET THEN
                   REPEAT
                     Rec := Company;
                     INSERT;
                   UNTIL Company.NEXT = 0;

                 MarkAll;
               END;

    OnAfterGetRecord=BEGIN
                       Selected := SelectedCompany.GET(Name);
                     END;

    OnQueryClosePage=BEGIN
                       FileName := Description;
                       IF CloseAction = ACTION::OK THEN BEGIN
                         IF EXPORTDATA(
                              TRUE,
                              FileName,
                              Description,
                              IncludeApplication,
                              IncludeApplicationData,
                              IncludeGlobalData,
                              SelectedCompany)
                         THEN BEGIN
                           MESSAGE(CompletedMsg);
                           EXIT(TRUE)
                         END;
                         EXIT(FALSE)
                       END;

                       EXIT(TRUE);
                     END;

  }
  CONTROLS
  {
    { 1   ;0   ;Container ;
                ContainerType=ContentArea }

    { 2   ;1   ;Group     ;
                CaptionML=[ENU=General;
                           ENG=General];
                GroupType=Group }

    { 3   ;2   ;Field     ;
                CaptionML=[ENU=Description;
                           ENG=Description];
                SourceExpr=Description }

    { 11  ;2   ;Group     ;
                CaptionML=[ENU=Export;
                           ENG=Export];
                GroupType=Group }

    { 12  ;3   ;Field     ;
                CaptionML=[ENU=All Companies;
                           ENG=All Companies];
                SourceExpr=IncludeAllCompanies;
                OnValidate=BEGIN
                             MarkAll;
                           END;
                            }

    { 6   ;3   ;Field     ;
                CaptionML=[ENU=Global Data;
                           ENG=Global Data];
                SourceExpr=IncludeGlobalData }

    { 4   ;3   ;Field     ;
                CaptionML=[ENU=Application Data;
                           ENG=Application Data];
                SourceExpr=IncludeApplicationData }

    { 5   ;3   ;Field     ;
                CaptionML=[ENU=Application;
                           ENG=Application];
                SourceExpr=IncludeApplication }

    { 8   ;2   ;Group     ;
                GroupType=Repeater }

    { 10  ;3   ;Field     ;
                CaptionML=[ENU=Export;
                           ENG=Export];
                SourceExpr=Selected;
                OnValidate=BEGIN
                             IF Selected THEN BEGIN
                               SelectedCompany := Rec;
                               IF SelectedCompany.INSERT THEN;
                             END ELSE BEGIN
                               IncludeAllCompanies := FALSE;
                               IF SelectedCompany.GET(Name) THEN
                                 SelectedCompany.DELETE;
                             END;
                           END;
                            }

    { 9   ;3   ;Field     ;
                Width=30;
                CaptionML=[ENU=Company Name;
                           ENG=Company Name];
                SourceExpr=Name }

  }
  CODE
  {
    VAR
      SelectedCompany@1002 : TEMPORARY Record 2000000006;
      FileName@1007 : Text;
      Description@1000 : Text;
      IncludeApplication@1003 : Boolean;
      IncludeApplicationData@1004 : Boolean;
      IncludeGlobalData@1006 : Boolean;
      IncludeAllCompanies@1005 : Boolean;
      Selected@1001 : Boolean;
      CompletedMsg@1008 : TextConst 'ENU=The data was exported successfully.;ENG=The data was exported successfully.';

    LOCAL PROCEDURE MarkAll@1();
    BEGIN
      SelectedCompany.DELETEALL;
      IF IncludeAllCompanies THEN BEGIN
        IF FINDSET THEN
          REPEAT
            SelectedCompany := Rec;
            SelectedCompany.INSERT;
          UNTIL NEXT = 0;
      END;

      CurrPage.UPDATE(FALSE);
    END;

    BEGIN
    END.
  }
}
