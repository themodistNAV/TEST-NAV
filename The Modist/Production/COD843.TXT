OBJECT Codeunit 843 Cash Flow Wksh. - Register
{
  OBJECT-PROPERTIES
  {
    Date=07/09/12;
    Time=12:00:00 PM;
    Version List=NAVW17.00;
  }
  PROPERTIES
  {
    TableNo=846;
    OnRun=BEGIN
            CFWkshLine.COPY(Rec);
            Code;
            COPY(CFWkshLine);
          END;

  }
  CODE
  {
    VAR
      Text1001@1001 : TextConst 'ENU=Do you want to register the worksheet lines?;ENG=Do you want to register the worksheet lines?';
      Text1002@1002 : TextConst 'ENU=There is nothing to register.;ENG=There is nothing to register.';
      Text1003@1003 : TextConst 'ENU=The worksheet lines were successfully registered.;ENG=The worksheet lines were successfully registered.';
      CFWkshLine@1007 : Record 846;

    LOCAL PROCEDURE Code@1000();
    BEGIN
      WITH CFWkshLine DO BEGIN
        IF NOT CONFIRM(Text1001) THEN
          EXIT;

        CODEUNIT.RUN(CODEUNIT::"Cash Flow Wksh.-Register Batch",CFWkshLine);

        IF "Line No." = 0 THEN
          MESSAGE(Text1002)
        ELSE
          MESSAGE(Text1003);

        IF NOT FIND('=><') THEN BEGIN
          RESET;
          "Line No." := 1;
        END;
      END;
    END;

    BEGIN
    END.
  }
}
