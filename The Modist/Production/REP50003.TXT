    LOCAL PROCEDURE FindApplnEntriesDtldtLedgEntry@1(pVendorLedgerEntry@1000000000 : Record 25;pPayment@1000000001 : Boolean);
    VAR
      DtldVendLedgEntry1@1000 : Record 380;
      DtldVendLedgEntry2@1001 : Record 380;
    BEGIN
      DtldVendLedgEntry1.SETCURRENTKEY("Vendor Ledger Entry No.");
      DtldVendLedgEntry1.SETRANGE("Vendor Ledger Entry No.",pVendorLedgerEntry."Entry No.");
      DtldVendLedgEntry1.SETRANGE(Unapplied,FALSE);
      IF DtldVendLedgEntry1.FINDSET THEN
        REPEAT
          IF DtldVendLedgEntry1."Vendor Ledger Entry No." =
             DtldVendLedgEntry1."Applied Vend. Ledger Entry No."
          THEN BEGIN
            DtldVendLedgEntry2.INIT;
            DtldVendLedgEntry2.SETCURRENTKEY("Applied Vend. Ledger Entry No.","Entry Type");
            DtldVendLedgEntry2.SETRANGE(
              "Applied Vend. Ledger Entry No.",DtldVendLedgEntry1."Applied Vend. Ledger Entry No.");
            DtldVendLedgEntry2.SETRANGE("Entry Type",DtldVendLedgEntry2."Entry Type"::Application);
            DtldVendLedgEntry2.SETRANGE(Unapplied,FALSE);
            IF DtldVendLedgEntry2.FINDSET THEN
              REPEAT
                IF DtldVendLedgEntry2."Vendor Ledger Entry No." <>
                   DtldVendLedgEntry2."Applied Vend. Ledger Entry No."
                THEN BEGIN
                  AppliedVendorLedgerEntry.SETCURRENTKEY("Entry No.");
                  AppliedVendorLedgerEntry.SETRANGE("Entry No.",DtldVendLedgEntry2."Vendor Ledger Entry No.");
                  IF pPayment THEN
                    AppliedVendorLedgerEntry.SETRANGE("Document Type",AppliedVendorLedgerEntry."Document Type"::Payment)
                  ELSE
                    AppliedVendorLedgerEntry.SETRANGE("Document Type",AppliedVendorLedgerEntry."Document Type"::Refund);
                  IF AppliedVendorLedgerEntry.FIND('-') THEN BEGIN
                    TmpAppliedVendorLedgerEntry.INIT;
                    TmpAppliedVendorLedgerEntry := AppliedVendorLedgerEntry;
                    TmpAppliedVendorLedgerEntry."Closed by Amount" := AppliedVendorLedgerEntry.Amount;
                    IF TmpAppliedVendorLedgerEntry.INSERT THEN;
                  END;
                END;
              UNTIL DtldVendLedgEntry2.NEXT = 0;
          END ELSE BEGIN
            AppliedVendorLedgerEntry.SETCURRENTKEY("Entry No.");
            AppliedVendorLedgerEntry.SETRANGE("Entry No.",DtldVendLedgEntry1."Applied Vend. Ledger Entry No.");
            IF pPayment THEN
              AppliedVendorLedgerEntry.SETRANGE("Document Type",AppliedVendorLedgerEntry."Document Type"::Payment)
            ELSE
              AppliedVendorLedgerEntry.SETRANGE("Document Type",AppliedVendorLedgerEntry."Document Type"::Refund);
            IF AppliedVendorLedgerEntry.FIND('-') THEN BEGIN
              TmpAppliedVendorLedgerEntry.INIT;
              TmpAppliedVendorLedgerEntry := AppliedVendorLedgerEntry;
              TmpAppliedVendorLedgerEntry."Closed by Amount" := AppliedVendorLedgerEntry.Amount;
              IF TmpAppliedVendorLedgerEntry.INSERT THEN;
            END;
          END;
        UNTIL DtldVendLedgEntry1.NEXT = 0;
    END;

    BEGIN
    {
      //Use RecRef better
      //DOC TM3021 ST 26/08/2018 - Code added to calculate indirect inv. amt.
    }
    END.
