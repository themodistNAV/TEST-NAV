    LOCAL PROCEDURE SetEnabled@1();
    BEGIN
      MandatoryLanguagesEnabled := "Check Missing Captions";
      ErrorLockedByOtherEnabled := NOT ErrorNotLocked;

      SMTPFieldsEnabled := "Mail Type" = "Mail Type"::SMTP;
      AuthenticationFieldsEnabled := ("Mail Type" = "Mail Type"::SMTP) AND Authentication;

      DevelopingFieldsEditable := Editor = Editor::Navgate;
    END;

    LOCAL PROCEDURE NoOfOptionsChecked@2(CheckType@1000 : Integer) : Integer;
    VAR
      CheckGuidelinesMgt@1001 : Codeunit 11102060;
    BEGIN
      EXIT(CheckGuidelinesMgt.NoOfOptionsChecked(Rec, CheckType));
    END;

    LOCAL PROCEDURE NoOfCriticalChecks@3() ExitValue : Integer;
    VAR
      i@1000 : Integer;
    BEGIN
      IF "Critical Guideline Checks" <> '' THEN
        ExitValue := 1;

      FOR i := 1 TO STRLEN("Critical Guideline Checks") DO
        IF "Critical Guideline Checks"[i] = '|' THEN
          ExitValue += 1;
    END;

    LOCAL PROCEDURE SetLockBooleans@4();
    BEGIN
      ErrorNotLocked := "SQL Check Object Lock Type" IN ["SQL Check Object Lock Type"::"Error if Not Locked"];

      ErrorLockedByOther :=
        "SQL Check Object Lock Type" IN [
          "SQL Check Object Lock Type"::"Error if Locked by Other",
          "SQL Check Object Lock Type"::"Error if Not Locked"];
    END;

    LOCAL PROCEDURE ValidateLockBooleans@5();
    BEGIN
      IF ErrorNotLocked THEN
        VALIDATE("SQL Check Object Lock Type", "SQL Check Object Lock Type"::"Error if Not Locked")
      ELSE
        IF ErrorLockedByOther THEN
          VALIDATE("SQL Check Object Lock Type", "SQL Check Object Lock Type"::"Error if Locked by Other")
        ELSE
          VALIDATE("SQL Check Object Lock Type", "SQL Check Object Lock Type"::None);
    END;

    LOCAL PROCEDURE SetStyles@6();
    VAR
      Mgt@1000 : Codeunit 11102035;
      ObjectMgt@1001 : Codeunit 11102046;
      FileHandler@1002 : Codeunit 11102097;
      ApplicationSettings@1003 : Codeunit 11102101;
      EditorMgt@1004 : Codeunit 11102106;
      ctDatabaseCode@1005 : TextConst 'DEU=<%1: %2>;ENU=<%1: %2>;ESP=<%1: %2>;FRA=<%1: %2>;NLD=<%1: %2>;DEA=<%1: %2>';
      ctNone@1006 : TextConst 'DEU=<None>;ENU=<None>;ESP=<None>;FRA=<None>;NLD=<Geen>;DEA=<None>';
      ctWorkingDirectory@1007 : TextConst 'DEU=<%1>;ENU=<%1>;ESP=<%1>;FRA=<%1>;NLD=<%1>;DEA=<%1>';
    BEGIN
      IF NOT Mgt.CanRunDotNet THEN BEGIN
        WorkingDirectoryEnabled := FALSE;
        WorkingDirectory := ctNone;
        WorkingDirectoryStyle := 'Subordinate';
      END ELSE BEGIN
        WorkingDirectoryEnabled := TRUE;
        IF "Working Directory" = '' THEN BEGIN
          IF "Shared Path" THEN
            WorkingDirectory := STRSUBSTNO(ctWorkingDirectory, TEMPORARYPATH)
          ELSE
            WorkingDirectory := STRSUBSTNO(ctWorkingDirectory, FileHandler.ClientTempDirectory);
          WorkingDirectoryStyle := 'Subordinate';
        END ELSE BEGIN
          WorkingDirectory := "Working Directory";
          WorkingDirectoryStyle := '';
        END;
      END;

      IF "Database Code" = '' THEN BEGIN
        DatabaseCode := STRSUBSTNO(ctDatabaseCode, ApplicationSettings.ServerInstanceName, ApplicationSettings.DatabaseName);
        DatabaseCodeStyle := 'Subordinate';
      END ELSE BEGIN
        DatabaseCode := "Database Code";
        DatabaseCodeStyle := '';
      END;

      ProjectTagDocTrigger2 := ProjectTagDocTrigger;
      IF "Project Tag Doc. Trigger" = '' THEN
        ProjectTagDocTriggerStyle := 'Subordinate'
      ELSE
        ProjectTagDocTriggerStyle := '';

      ModificationTag2 := ModificationTag;
      IF "Modification Tag" = '' THEN
        ModificationTagStyle := 'Subordinate'
      ELSE
        ModificationTagStyle := '';

      ModificationStartTag2 := ModificationStartTag;
      IF "Modification Start Tag" = '' THEN
        ModificationStartTagStyle := 'Subordinate'
      ELSE
        ModificationStartTagStyle := '';

      ModificationEndTag2 := ModificationEndTag;
      IF "Modification End Tag" = '' THEN
        ModificationEndTagStyle := 'Subordinate'
      ELSE
        ModificationEndTagStyle := '';

      IF "Snippet Folder" = '' THEN BEGIN
        SnippetFolder := EditorMgt.SnippetFolder;
        SnippetFolderStyle := 'Subordinate';
      END ELSE BEGIN
        SnippetFolder := "Snippet Folder";
        SnippetFolderStyle := '';
      END;

      IF "Specification File" = '' THEN BEGIN
        SpecificationFile := EditorMgt.SpecificationFile;
        SpecificationFileStyle := 'Subordinate';
      END ELSE BEGIN
        SpecificationFile := "Specification File";
        SpecificationFileStyle := '';
      END;

      IF "Default UID Offset" = 0 THEN BEGIN
        UidOffset := DbUidOffset;
        UidOffsetStyle := 'Subordinate';
      END ELSE BEGIN
        UidOffset := "Default UID Offset";
        UidOffsetStyle := '';
      END;

      IF "Object Cache (MB)" = 0 THEN BEGIN
        ObjectCacheMB := 500;
        ObjectCacheMBStyle := 'Subordinate';
      END ELSE BEGIN
        ObjectCacheMB := "Object Cache (MB)";
        ObjectCacheMBStyle := '';
      END;

      SmtpPort2 := SMTPPort;
      IF "SMTP Port" = 0 THEN
        SmtpPortStyle := 'Subordinate'
      ELSE
        SmtpPortStyle := '';

      ObjectFileNameFormat2 := ObjectFileNameFormat;
      IF "Object File Name Format" = '' THEN
        ObjectFileNameFormatStyle := 'Subordinate'
      ELSE
        ObjectFileNameFormatStyle := '';

      IF "Command Line Parameters" = '' THEN BEGIN
        CommandLineParameters := ObjectMgt.DefaultCommandParameters(TRUE);
        CommandLineParametersStyle := 'Subordinate';
      END ELSE BEGIN
        CommandLineParameters := "Command Line Parameters";
        CommandLineParametersStyle := '';
      END;
    END;
