                                  SubSeasonCode := '';
                                  BrandName := '';
                                  CurrencyCode := '';
                                  CASE "Document Type" OF
                                    "Document Type"::Order:
                                      BEGIN
                                        IF PurchaseHeader."No." <> "Document No." THEN
                                          IF NOT PurchaseHeader.GET("Document Type","Document No.") THEN
                                            CurrReport.SKIP;
                                        PurchaseLine.SETRANGE("Document Type",PurchaseHeader."Document Type");
                                        PurchaseLine.SETRANGE("Document No.",PurchaseHeader."No.");
                                        PurchaseLine.SETFILTER("Brand Code",'<>%1','');
                                        IF PurchaseLine.FINDFIRST THEN
                                          BrandName := PurchaseLine."Brand Code";
                                        VendorName := PurchaseHeader."Pay-to Name";
                                        SeasonCode := PurchaseHeader."Season Code";
                                        SubSeasonCode := PurchaseHeader."Sub-Season Code";
                                        VendorNo := PurchaseHeader."Pay-to Vendor No.";
                                        CurrencyCode := PurchaseHeader."Currency Code";
                                        PaymentTerms := PurchaseHeader."Payment Terms Code";
                                      END;
                                  END;
                                END;
